// Global Interactive Reaction Trigger Helper
window.triggerTapReaction = function(msg, isJar = false) {
    const feedback = document.getElementById('interaction-feedback');
    if(feedback) {
        feedback.innerText = msg;
        feedback.classList.add('active');
    }
    if(isJar) {
        const jar = document.getElementById('jar-target');
        if(jar) {
            jar.classList.add('bumped');
            setTimeout(() => jar.classList.remove('bumped'), 4000);
        }
    }
}

window.animateJar = function() {
    triggerTapReaction("💖 You are appreciated more than you know.", true);
}

// Master Render Engine Architecture
function assembleSlide() {
    const slideData = experienceMap[currentStep];
    const stage = document.getElementById('slide-stage-content');
    const feedback = document.getElementById('interaction-feedback');
    
    // Reset feedback text box state
    feedback.classList.remove('active');
    
    // UI Out-Animation Phase
    stage.style.opacity = '0';
    stage.style.transform = 'translateY(-10px)';
    
    setTimeout(() => {
        // Construct and Inject Content HTML
        let output = `
            <h3>${slideData.subtitle}</h3>
            <h2>${slideData.title}</h2>
            ${slideData.html()}
            <div class="choices-wrapper">
                <button class="btn-choice" onclick="processChoice(1)">${slideData.btn1}</button>
                <button class="btn-choice" onclick="processChoice(2)">${slideData.btn2}</button>
            </div>
        `;
        
        stage.innerHTML = output;
        
        // Update Progress bar parameters
        const totalSlides = Object.keys(experienceMap).length;
        const widthPercentage = (currentStep / totalSlides) * 100;
        document.getElementById('progress-indicator').style.width = `${widthPercentage}%`;
        
        // UI In-Animation Phase
        stage.style.transition = 'all 0.4s ease';
        stage.style.opacity = '1';
        stage.style.transform = 'translateY(0)';
    }, 200);
}

// Handle Choice Execution & Advanced Logic Paths
window.processChoice = function(choiceId) {
    const slideData = experienceMap[currentStep];
    const reactionText = choiceId === 1 ? slideData.react1 : slideData.react2;
    
    // Display targeted unique action text
    triggerTapReaction(reactionText);
    
    // Execute slide-specific cleanup callbacks if defined
    if (slideData.onLeave) {
        slideData.onLeave(choiceId);
    }
    
    // Special high-tier celebrations logic
    if (currentStep === 6 && choiceId === 1) {
        spawnInstantExplosion('🎉');
    }
    if (currentStep === 11) {
        spawnInstantExplosion('💖');
    }
    
    // Lock controls briefly for user to ingest reaction text, then advance
    setTimeout(() => {
        if (currentStep === 20) {
            if (choiceId === 1) {
                // Ignite grand finale sequence loop
                triggerGrandFinaleExplosions();
            } else {
                // Loop back around gracefully to slide 1
                currentStep = 1;
                customAesthetic = "pastel";
                document.body.style.background = 'linear-gradient(135deg, #fbc2eb 0%, #a6c1ee 100%)';
                document.documentElement.style.setProperty('--card-bg', 'rgba(255, 255, 255, 0.45)');
                document.documentElement.style.setProperty('--text-main', '#2d1b4e');
                document.documentElement.style.setProperty('--text-muted', '#624b7a');
                assembleSlide();
            }
        } else {
            currentStep++;
            assembleSlide();
            
            // Auto trigger intense canvas assets when landing on the ultimate screen
            if (currentStep === 20) {
                setTimeout(() => { triggerGrandFinaleExplosions(); }, 600);
            }
        }
    }, 2500);
}

/* ==========================================================================
   HIGH-PERFORMANCE CANVAS CELEBRATION ENGINE
   Manages beautiful ambient floating particles, stars, hearts, and fireworks
   ========================================================================== */
const canvas = document.getElementById('canvas-ambient');
const ctx = canvas.getContext('2d');

let particlesArray = [];
let fireworkArray = [];
let globalCelebrationActive = false;

function resizeCanvas() {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
}
window.addEventListener('resize', resizeCanvas);
resizeCanvas();

// Universal Particle Blueprint Class
class CelebrationParticle {
    constructor(x, y, char, size, speedX, speedY, color = null) {
        this.x = x;
        this.y = y;
        this.char = char;
        this.size = size;
        this.speedX = speedX;
        this.speedY = speedY;
        this.alpha = 1;
        this.decay = Math.random() * 0.015 + 0.005;
        this.color = color;
        this.rotation = Math.random() * Math.PI * 2;
        this.rotSpeed = (Math.random() - 0.5) * 0.05;
    }
    update() {
        this.x += this.speedX;
        this.y += this.speedY;
        this.rotation += this.rotSpeed;
        if (globalCelebrationActive) {
            this.speedY += 0.04; // Gravity simulation during finale
            this.alpha -= this.decay;
        } else {
            // Soft drift upwards for ambient mode
            if (this.y < -20) this.y = canvas.height + 20;
        }
    }
    draw() {
        ctx.save();
        ctx.globalAlpha = this.alpha;
        ctx.translate(this.x, this.y);
        ctx.rotate(this.rotation);
        if (this.color) {
            ctx.fillStyle = this.color;
            ctx.font = `${this.size}px serif`;
            ctx.fillText(this.char, 0, 0);
        } else {
            ctx.fillStyle = customAesthetic === 'pastel' ? '#624b7a' : '#c084fc';
            ctx.font = `${this.size}px serif`;
            ctx.fillText(this.char, 0, 0);
        }
        ctx.restore();
    }
}

// Generate base ambient movement values
function initAmbientParticles() {
    particlesArray = [];
    const pool = ['🌸', '✨', '💖', '✨'];
    for (let i = 0; i < 35; i++) {
        let x = Math.random() * canvas.width;
        let y = Math.random() * canvas.height;
        let char = pool[Math.floor(Math.random() * pool.length)];
        let size = Math.random() * 14 + 14;
        let speedX = (Math.random() - 0.5) * 0.6;
        let speedY = -(Math.random() * 0.5 + 0.3);
        particlesArray.push(new CelebrationParticle(x, y, char, size, speedX, speedY));
    }
}
initAmbientParticles();

function spawnInstantExplosion(emoji) {
    for (let i = 0; i < 24; i++) {
        let angle = Math.random() * Math.PI * 2;
        let speed = Math.random() * 5 + 2;
        let p = new CelebrationParticle(
            canvas.width / 2,
            canvas.height / 2,
            emoji,
            Math.random() * 16 + 16,
            Math.cos(angle) * speed,
            Math.sin(angle) * speed
        );
        p.alpha = 1;
        p.decay = 0.02;
        fireworkArray.push(p);
    }
}

function launchFirework() {
    const targetX = Math.random() * canvas.width;
    const targetY = Math.random() * (canvas.height * 0.6);
    const colors = ['✨', '🌸', '💖', '🎉', '❤️', '🌟', '🦄'];
    const chosenEmoji = colors[Math.floor(Math.random() * colors.length)];
    
    for (let i = 0; i < 30; i++) {
        let angle = Math.random() * Math.PI * 2;
        let speed = Math.random() * 6 + 1;
        let p = new CelebrationParticle(
            targetX,
            targetY,
            chosenEmoji,
            Math.random() * 18 + 14,
            Math.cos(angle) * speed,
            Math.sin(angle) * speed
        );
        fireworkArray.push(p);
    }
}

let finaleInterval;
function triggerGrandFinaleExplosions() {
    globalCelebrationActive = true;
    launchFirework();
    launchFirework();
    
    if (finaleInterval) clearInterval(finaleInterval);
    finaleInterval = setInterval(() => {
        if (currentStep === 20 && globalCelebrationActive) {
            launchFirework();
        } else {
            clearInterval(finaleInterval);
            globalCelebrationActive = false;
        }
    }, 700);
}

// Master RequestAnimationFrame Loop Matrix
function renderEngineLoop() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    // Draw Ambient Layer
    for (let i = 0; i < particlesArray.length; i++) {
        particlesArray[i].update();
        particlesArray[i].draw();
    }
    
    // Draw Active Burst/Explosion Layer
    for (let i = fireworkArray.length - 1; i >= 0; i--) {
        fireworkArray[i].update();
        fireworkArray[i].draw();
        if (fireworkArray[i].alpha <= 0) {
            fireworkArray.splice(i, 1);
        }
    }
    
    requestAnimationFrame(renderEngineLoop);
}
requestAnimationFrame(renderEngineLoop);

// Initialize System Boot sequences
assembleSlide();
